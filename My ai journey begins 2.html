def print_personalized_welcome():

    """

    This function prompts the user for their name and then prints a friendly,

    personalized welcome message to the console.


    It demonstrates basic input from the user and output to the console.

    """

    # Ask the user to input their name.

    # The input() function will display the string passed to it and wait

    # for the user to type something and press Enter.

    user_name = input("Please enter your name: ")


    # Create the personalized welcome message.

    # An f-string (formatted string literal) is used here for easy

    # embedding of variables (like user_name) directly into the string.

    welcome_message = f"Hello, {user_name}! Welcome and have a fantastic time!"


    # Display the final welcome message to the user.

    print(welcome_message)


# Call the function to start the script.

# This line ensures that when the Python file is executed, the

# 'print_personalized_welcome' function is called, initiating the interaction.

print_personalized_welcome()

    <script>

        const colors = [

            { bg: 'bg-blue-200', text: 'text-blue-800', border: 'border-blue-300' },

            { bg: 'bg-purple-200', text: 'text-purple-800', border: 'border-purple-300' },

            { bg: 'bg-green-200', text: 'text-green-800', border: 'border-green-300' },

            { bg: 'bg-orange-200', text: 'text-orange-800', border: 'border-orange-300' },

            { bg: 'bg-pink-200', text: 'text-pink-800', border: 'border-pink-300' },

            { bg: 'bg-teal-200', text: 'text-teal-800', border: 'border-teal-300' }

        ];


        function createAiSign() {

            const sign = document.createElement('div');

            const randomColor = colors[Math.floor(Math.random() * colors.length)];

            const randomDuration = Math.random() * 15 + 10; // 10 to 25 seconds

            const randomDelay = Math.random() * -10; // -10 to 0 seconds, so some start already in motion

            const randomTop = Math.random() * (document.body.clientHeight * 0.9) + (document.body.clientHeight * 0.05); // 5% to 95% of body height


            sign.textContent = "My AI Journey Begins";

            sign.className = `ai-sign ${randomColor.bg} ${randomColor.text} ${randomColor.border} border-2`;

            sign.style.top = `${randomTop}px`;

            sign.style.animationDuration = `${randomDuration}s`;

            sign.style.animationDelay = `${randomDelay}s`;

            sign.style.left = '100vw'; // Start off-screen right


            document.body.appendChild(sign);


            sign.addEventListener('animationend', () => {

                sign.remove(); // Remove element after animation to clean up DOM

                createAiSign(); // Create a new one

            });

        }


        // Initial creation of signs, staggered

        for (let i = 0; i < 15; i++) {

            setTimeout(createAiSign, i * 700);

        }


        // Re-create signs on window resize to adjust positions

        window.addEventListener('resize', () => {

            document.querySelectorAll('.ai-sign').forEach(sign => sign.remove());

            for (let i = 0; i < 15; i++) {

                setTimeout(createAiSign, i * 700);

            }

        });


        // Gemini API Integration for AI Project Idea Generation

        document.getElementById('generateIdeaBtn').addEventListener('click', async () => {

            const ideaOutputDiv = document.getElementById('ideaOutput');

            const ideaTextP = document.getElementById('ideaText');


            ideaOutputDiv.classList.remove('hidden');

            ideaTextP.textContent = "Generating a fresh idea... please wait.";

            ideaOutputDiv.classList.remove('bg-red-50', 'border-red-200', 'text-red-800'); // Clear error styling

            ideaOutputDiv.classList.add('bg-blue-50', 'border-blue-200', 'text-blue-800'); // Set loading styling


            const prompt = `Given the discussions about reasons to study AI (career, innovation, skill, versatility, understanding future), foundational concepts (planetary water, photosynthesis, speed of light, human brain development), version control with GitHub, and AI interaction philosophy, generate one concise and creative new AI project idea (under 50 words). Focus on a practical application or an innovative concept.`;


            let chatHistory = [];

            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };

            const apiKey = ""; // Canvas will automatically provide this in runtime


            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


            try {

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const result = await response.json();


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    ideaTextP.textContent = text;

                    ideaOutputDiv.classList.remove('bg-blue-50', 'border-blue-200', 'text-blue-800'); // Clear loading styling

                    ideaOutputDiv.classList.add('bg-green-50', 'border-green-200', 'text-green-800'); // Set success styling

                } else {

                    ideaTextP.textContent = "Failed to generate idea. Please try again.";

                    ideaOutputDiv.classList.remove('bg-blue-50', 'border-blue-200', 'text-blue-800'); // Clear loading styling

                    ideaOutputDiv.classList.add('bg-red-50', 'border-red-200', 'text-red-800'); // Set error styling

                }

            } catch (error) {

                console.error("Error calling Gemini API:", error);

                ideaTextP.textContent = "Error generating idea. Check console for details.";

                ideaOutputDiv.classList.remove('bg-blue-50', 'border-blue-200', 'text-blue-800'); // Clear loading styling

                ideaOutputDiv.classList.add('bg-red-50', 'border-red-200', 'text-red-800'); // Set error styling

            }

        });

    </script>
